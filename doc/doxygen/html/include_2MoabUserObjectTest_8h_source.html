180  bool getCentroid(std::shared_ptr<moab::Interface> moabPtr,
181  moab::EntityHandle elem,
182  Point & centroid,
183  std::string& errmsg){
184 
185  moab::ErrorCode rval;
186 
187  // Fetch elem connectivity
188  std::vector<moab::EntityHandle> conn;
189  rval = moabPtr->get_connectivity(&elem, 1, conn);
190  if (rval != moab::MB_SUCCESS){
191  errmsg="Failed to get connectivity.";
192  return false;
193  }
194 
195  // Fetch vertex coords
196  size_t nNodes = conn.size();
197  if(nNodes==0){
198  errmsg="Failed to get nodes.";
199  return false;
200  }
201 
202  std::vector<double> nodeCoords(nNodes*3);
203  rval = moabPtr->get_coords(conn.data(), nNodes, nodeCoords.data());
204  if (rval != moab::MB_SUCCESS){
205  errmsg="Failed to get nodes' coords.";
206  return false;
207  }
208 
209  // Compute the centroid of the element vertices
210  centroid = Point(0.,0.,0.);
211  for(size_t inode=0; inode<nNodes; ++inode){
212  for(int i=0; i<3; i++){
213  double coord = nodeCoords.at(3*inode +i);
214  centroid(i)+= coord /double(nNodes);
215  }
216  }
217 
218  return true;
219  }

265  double elemRadius(Elem& el){
266 
267  Point centroid(0.,0.,0.);
268  unsigned int nNodes = el.n_nodes();
269  for(unsigned int iNode=0; iNode<nNodes; ++iNode){
270  // Get the point coords for this node
271  const Point& point = el.point(iNode);
272  centroid += point;
273  }
274  centroid /= double(nNodes);
275  // Scale to same units as Moab
276  centroid *= lengthscale;
277  return centroid.norm();

358

361 
362  // Loop over the elements
363  auto itelem = mesh.elements_begin();
364  auto endelem = mesh.elements_end();
365  size_t iElem=0;

372  ASSERT_LT(soln_index,solsize);

376 
377  // Get the solution we expect
378  double solExpectNow=solExpect.at(iElem);
379 
380  // Compare
381  double solDiff = fabs(sol-solExpectNow);
382  EXPECT_LT(solDiff,tol);
